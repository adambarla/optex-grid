\newdimen\h 
\newdimen\w 
\newdimen\curr 
\newcount\linecount 
\newcount\blockcount
%/ horizontal spacing size, vertical spacing size(left margin, top margin,)
\def\grid#1,#2(#3,#4,#5,#6){% defining implicit parameters
  \if$#1$ \def\x{1mm} \else \def\x{#1} \fi
  \if$#2$ \def\y{1mm} \else \def\y{#2} \fi
  \if$#3$ \def\left{\hoffset} \else \def\left{#3} \fi
  \if$#4$ \def\top{\voffset} \else \def\top{#4} \fi
  \if$#5$ \w=\hsize \else \w=#5 \fi
  \if$#6$ \h=\vsize \else \h=#6 \fi
  \_pgbackground={\Green
    \blockcount=0 \linecount=0 \curr=0pt\vertical
    \blockcount=0 \linecount=0 \curr=0pt\horizontal}%
}
\def\vertical {\noindent
  \ifnum\curr<\w
    \newcount\temp \temp=\blockcount \multiply\temp10
    \hbox to0pt{\hskip\left\hskip\curr\vbox to0pt{\vskip\top\noindent
      \ifnum\linecount =\temp
        \global\advance\blockcount1
        \newdimen\largerline \largerline=\h \advance\largerline\baselineskip
        \vbox to0pt{\vss\noindent\hskip2pt\the\linecount\vskip1pt}%
        \vskip-\baselineskip\noindent\vrule height\largerline width 1pt
      \else
        \vrule height\h width 0.1pt
      \fi
    \vss}\hss}%
    \advance\linecount1
    \advance\curr\x
    \vertical
  \fi
}
\def\horizontal {\noindent
  \ifnum\curr<\h
    \newcount\temp \temp=\blockcount \multiply\temp10
    \hbox to0pt{\hskip\left \vbox to0pt{\vskip\top\vskip\curr
      \ifnum\linecount =\temp
        \global\advance\blockcount1
        \newdimen\largerline \largerline=\w \advance\largerline\baselineskip
        \hbox to0pt{\hskip-\baselineskip\vbox to0pt{\hrule width\largerline height 1pt\vss}\hss}%
        \vbox to0pt{\vskip-\baselineskip\vskip2pt\hbox to0pt{\hss\noindent\the\linecount\hskip1pt}\vss}%
      \else
        \hrule width\w height 0.1pt
      \fi
    \vss}\hss}%
    \advance\linecount1
    \advance\curr\y
    \horizontal
  \fi
}
